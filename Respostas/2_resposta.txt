CREATE DATABASE FILMES_EM_ACAO;



CREATE TABLE PRICES(
	
	ID SERIAL PRIMARY KEY,
	CATEGORIE VARCHAR(255) UNIQUE NOT NULL,
	VALUE DECIMAL NOT NULL

);


CREATE TABLE MOVIES(

	ID SERIAL PRIMARY KEY,
	NAME VARCHAR(255) UNIQUE NOT NULL,
	ID_PRICES SERIAL,
    FOREIGN KEY (ID_PRICES) REFERENCES PRICES (ID)

);
